## 5. Pseudocode
### Order Service: placeOrder(customerID, restaurantID, items, deliveryAddress)
```
FUNCTION placeOrder(customerID, restaurantID, items, deliveryAddress)
  VALIDATE customerID, restaurantID, items, deliveryAddress
  CALCULATE totalAmount FROM items
  CREATE newOrder OBJECT WITH customerID, restaurantID, items, deliveryAddress, totalAmount, status="PENDING"
  SAVE newOrder TO OrderDatabase
  SEND "OrderPlaced" EVENT WITH orderID, restaurantID, totalAmount
  RETURN newOrder.orderID
END FUNCTION
```

### Restaurant & Menu Service: addMenuItem(restaurantID, itemDetails)
```
FUNCTION addMenuItem(restaurantID, itemDetails)
  VALIDATE restaurantID, itemDetails
  RETRIEVE restaurantProfile FROM RestaurantDatabase WHERE ID = restaurantID
  IF restaurantProfile IS NULL THEN
    RETURN ERROR "Restaurant not found"
  END IF
  CREATE newMenuItem OBJECT FROM itemDetails WITH uniqueItemID
  ADD newMenuItem TO restaurantProfile.menuItems
  UPDATE restaurantProfile IN RestaurantDatabase
  RETURN newMenuItem.itemID
END FUNCTION
```

### User & Profile Service: authenticateUser(username, password)
```
FUNCTION authenticateUser(username, password)
  RETRIEVE userProfile FROM UserDatabase WHERE username = username
  IF userProfile IS NULL THEN
    RETURN FALSE, "Invalid username or password"
  END IF
  HASH inputPassword USING userProfile.salt
  IF HASHED inputPassword MATCHES userProfile.hashedPassword THEN
    GENERATE authSessionToken FOR userProfile.userID
    RETURN TRUE, authSessionToken
  ELSE
    RETURN FALSE, "Invalid username or password"
  END IF
END FUNCTION
```

### Delivery Service: assignDeliveryAgent(orderID, restaurantLocation, customerLocation)
```
FUNCTION assignDeliveryAgent(orderID, restaurantLocation, customerLocation)
  RETRIEVE availableAgents FROM AgentDatabase NEAR restaurantLocation
  IF availableAgents IS EMPTY THEN
    RETURN ERROR "No agents available"
  END IF
  SELECT bestAgent FROM availableAgents BASED ON proximity AND currentLoad
  UPDATE orderStatus TO "ASSIGNED" IN OrderService VIA API
  UPDATE agentStatus TO "BUSY" FOR bestAgent
  NOTIFY bestAgent OF newAssignment(orderID, restaurantLocation, customerLocation)
  RETURN bestAgent.agentID
END FUNCTION
```

### Payment Service: processPayment(orderID, customerID, amount, paymentMethodDetails)
```
FUNCTION processPayment(orderID, customerID, amount, paymentMethodDetails)
  VALIDATE orderID, customerID, amount, paymentMethodDetails
  INITIALIZE paymentGatewayClient
  ATTEMPT transaction WITH paymentGatewayClient(amount, paymentMethodDetails)
  IF transaction SUCCESSFUL THEN
    RECORD paymentDetails IN PaymentDatabase WITH status="COMPLETED"
    UPDATE orderStatus TO "PAID" IN OrderService VIA API
    RETURN TRUE, "Payment successful"
  ELSE
    RECORD paymentDetails IN PaymentDatabase WITH status="FAILED"
    RETURN FALSE, "Payment failed"
  END IF
END FUNCTION
```
